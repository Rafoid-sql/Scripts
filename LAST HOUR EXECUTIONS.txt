SELECT 
	DSN.BEGIN_INTERVAL_TIME,
	DSQ.EXECUTIONS_DELTA,
	DU.USERNAME,
	DSQ.SQL_ID,
	DST.SQL_TEXT 
FROM 
	DBA_HIST_SQLSTAT DSQ
JOIN DBA_HIST_SNAPSHOT DSN ON DSQ.SNAP_ID=DSN.SNAP_ID
JOIN DBA_USERS DU ON DU.USER_ID=DSQ.PARSING_SCHEMA_ID
JOIN DBA_HIST_SQLTEXT DST ON DSQ.SQL_ID =DST.SQL_ID
WHERE 
	DSQ.SQL_ID = '&SQL_ID'
	AND BEGIN_INTERVAL_TIME > SYSDATE -1/24 
ORDER BY 1 DESC;
